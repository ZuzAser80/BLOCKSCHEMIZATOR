#include <iostream>
#include <string>
using namespace std;

int main() {
    setlocale(LC_ALL, "ru");
    system("chcp 1251");
    system("cls");
    int n;
    bool f = false;
    cout << "Input database size: ";
    cin >> n;
    string data[n][2];
    string s, temp, ans;
    cout << "Input database separated by spaces: " << endl;
    for (int i = 0; i < n; ++i) {
        cin >> data[i][0];
        cin >> data[i][1];
    }
    cout << "Input sentence: " << endl;
    cin >> s;
    for (int i = 0; i <= s.length(); ++i) {
        if (s[i] == '_' || s[i] == '\0') {
            f = false;
            for (int j = 0; j < n; j++) {
                if (temp == data[j][0]) {
                    ans += data[j][1] + "_";
                    f = true;
                }
            }
            if (!f) {
                ans += temp + "_";
            }
            temp = "";
        } else {
            temp += s[i];
        }
    }
    ans.pop_back();
    cout << "translated: " << ans;
}
